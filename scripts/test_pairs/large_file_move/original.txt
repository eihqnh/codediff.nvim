-- Utility functions module
local M = {}

function M.helper1()
    local x = 1
    local y = 2
    return x + y
end

function M.helper2()
    local data = {}
    for i = 1, 10 do
        table.insert(data, i * 2)
    end
    return data
end

function M.helper3()
    return os.time()
end

function M.helper4()
    local result = {}
    for k, v in pairs({ a = 1, b = 2, c = 3 }) do
        result[k] = v * 10
    end
    return result
end

function M.helper5()
    local sum = 0
    for i = 1, 100 do
        sum = sum + i
    end
    return sum
end

function M.helper6()
    local str = ""
    for i = 1, 5 do
        str = str .. tostring(i) .. ", "
    end
    return str
end

function M.helper7()
    local t = {}
    for i = 1, 20 do
        t[i] = math.random(1, 100)
    end
    table.sort(t)
    return t
end

function M.helper8()
    local count = 0
    for _ in pairs(M) do
        count = count + 1
    end
    return count
end

return M
